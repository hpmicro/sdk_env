690static inline hpm_stat_t i2c_set_data_count(I2C_Type *ptr, uint32_t size)
691{
692 if (size > I2C_SOC_TRANSFER_COUNT_MAX) {
693 return status_invalid_argument;
694 }
695#ifdef I2C_CTRL_DATACNT_HIGH_MASK
696 ptr->CTRL &= ~(I2C_CTRL_DATACNT_HIGH_MASK | I2C_CTRL_DATACNT_MASK);
697 ptr->CTRL |= I2C_CTRL_DATACNT_HIGH_SET(I2C_DATACNT_MAP(size) >> 8U) | I2C_CTRL_DATACNT_SET(I2C_DATACNT_MAP(size));
698#else
699 ptr->CTRL &= ~I2C_CTRL_DATACNT_MASK;
700 ptr->CTRL |= I2C_CTRL_DATACNT_SET(I2C_DATACNT_MAP(size));
701#endif
702 return status_success;
703}
