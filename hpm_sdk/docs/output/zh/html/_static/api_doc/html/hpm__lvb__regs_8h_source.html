12typedef struct {

14 __R uint8_t RESERVED0[12]; /* 0x4 - 0xF: Reserved */
15 __R uint32_t PHY_STAT; /* 0x10: LVDS TX PHY Status register */

17 struct {
18 __RW uint32_t CTL0; /* 0x1C: TX PHY Setting */

21} LVB_Type;
