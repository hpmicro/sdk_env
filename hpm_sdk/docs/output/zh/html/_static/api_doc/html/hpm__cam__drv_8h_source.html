133static inline uint32_t cam_get_pixel_format(display_pixel_format_t format)
134{
135 switch (format) {
136 case display_pixel_format_rgb565:
137 return CAM_COLOR_FORMAT_RGB565;
138 case display_pixel_format_yuv422:
139 case display_pixel_format_ycbcr422:
140 return CAM_COLOR_FORMAT_YCBCR422_YUV422;
141 case display_pixel_format_raw8:
142 return CAM_COLOR_FORMAT_RAW8;
143 default:
144 return CAM_COLOR_FORMAT_UNSUPPORTED;
145 }
146}
