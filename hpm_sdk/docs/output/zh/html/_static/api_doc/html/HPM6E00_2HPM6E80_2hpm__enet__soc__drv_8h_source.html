70static inline hpm_stat_t enet_rmii_enable_clock(ENET_Type *ptr, bool internal)
71{
72 hpm_stat_t stat = status_success;
73
74 /* use an internal PLL clock as reference clock for rmii mode */
75 if (ptr == HPM_ENET0) {
76 if (internal == true) {
77 /* use a pll clock */
78 ptr->CTRL2 |= ENET_CTRL2_ENET0_REFCLK_OE_MASK | ENET_CTRL2_ENET0_RMII_TXCLK_SEL_MASK;
79 } else {
80 /* use an external clock as reference clock for rmii mode */
81 ptr->CTRL2 |= ENET_CTRL2_ENET0_RMII_TXCLK_SEL_MASK; /* use an external clock */
82 }
83 } else {
84 return status_invalid_argument;
85 }
86
87 return stat;
88}
