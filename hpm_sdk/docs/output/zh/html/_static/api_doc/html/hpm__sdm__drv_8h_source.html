143static inline void sdm_enable_module_interrupt(SDM_Type *ptr, bool enable)
144{
145 if (enable) {
146 ptr->CTRL |= SDM_CTRL_IE_MASK;
147 } else {
148 ptr->CTRL &= ~SDM_CTRL_IE_MASK;
149 }

159static inline void sdm_enable_channel(SDM_Type *ptr, uint8_t ch_index, bool enable)
160{
161 if (enable) {
162 ptr->CTRL |= CHN_EN_MASK(ch_index);
163 } else {
164 ptr->CTRL &= ~CHN_EN_MASK(ch_index);
165 }
166}
