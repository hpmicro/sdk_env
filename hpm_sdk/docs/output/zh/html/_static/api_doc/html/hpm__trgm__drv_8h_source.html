105static inline void trgm_filter_set_filter_length(TRGM_Type *ptr, uint8_t input, uint32_t length)
106{
107#if defined(TRGM_SOC_HAS_FILTER_SHIFT) && TRGM_SOC_HAS_FILTER_SHIFT
108 uint32_t len = length;
109 uint8_t shift;
110 for (shift = 0; shift <= (TRGM_FILTCFG_FILTLEN_SHIFT_MASK >> TRGM_FILTCFG_FILTLEN_SHIFT_SHIFT); shift++) {
111 if (shift > 0) {
112 len >>= 1u;
113 }
114 if (len <= (TRGM_FILTCFG_FILTLEN_BASE_MASK >> TRGM_FILTCFG_FILTLEN_BASE_SHIFT)) {
115 break;
116 }
117 }
118 if (len > (TRGM_FILTCFG_FILTLEN_BASE_MASK >> TRGM_FILTCFG_FILTLEN_BASE_SHIFT)) {
119 len = (TRGM_FILTCFG_FILTLEN_BASE_MASK >> TRGM_FILTCFG_FILTLEN_BASE_SHIFT);
120 shift = (TRGM_FILTCFG_FILTLEN_SHIFT_MASK >> TRGM_FILTCFG_FILTLEN_SHIFT_SHIFT);
121 }
122 ptr->FILTCFG[input] = (ptr->FILTCFG[input] & ~(TRGM_FILTCFG_FILTLEN_BASE_MASK | TRGM_FILTCFG_FILTLEN_SHIFT_MASK))
123 | TRGM_FILTCFG_FILTLEN_BASE_SET(len) | TRGM_FILTCFG_FILTLEN_SHIFT_SET(shift);
124#else
125 ptr->FILTCFG[input] = (ptr->FILTCFG[input] & ~TRGM_FILTCFG_FILTLEN_MASK)
126 | TRGM_FILTCFG_FILTLEN_SET(length);
127#endif
128}

373#if defined(HPM_IP_FEATURE_TRGM_HRPWM_CALIBRATION_2) || defined(HPM_IP_FEATURE_TRGM_HRPWM_CALIBRATION_1)
381static inline void trgm_pwmv2_calibrate_delay_chain(TRGM_Type *trgm, trgm_pwmv2_calibration_mode_t *status)
382{
383#if defined(HPM_IP_FEATURE_TRGM_HRPWM_CALIBRATION_2)
384 uint16_t val_n, val_p;
385#endif
386 uint32_t delay_val;
387#if defined(HPM_IP_FEATURE_TRGM_HRPWM_CALIBRATION_1)
388 static uint16_t last_val = 0;
389 int16_t diff_val;
390#endif
391
392 switch (*status) {
393 case trgm_pwmv2_calibration_mode_begin:
394 trgm->PWM_CALIB_CFG &= ~TRGM_PWM_CALIB_CFG_CALIB_HW_ENABLE_MASK;
395 trgm->PWM_CALIB_CFG &= ~TRGM_PWM_CALIB_CFG_CALIB_SW_START_MASK;
396 trgm->PWM_CALIB_CFG |= TRGM_PWM_CALIB_CFG_CALIB_SW_START_MASK;
397 *status = trgm_pwmv2_calibration_mode_wait;
398 break;
399 case trgm_pwmv2_calibration_mode_wait:
400 if (TRGM_PWM_CALIB_STATUS0_CALIB_ON_GET(trgm->PWM_CALIB_STATUS0) == 0U) {
401#if defined(HPM_IP_FEATURE_TRGM_HRPWM_CALIBRATION_2)
402 val_n = TRGM_PWM_CALIB_STATUS1_CALIB_RESULT_N_GET(trgm->PWM_CALIB_STATUS1);
403 val_p = TRGM_PWM_CALIB_STATUS1_CALIB_RESULT_P_GET(trgm->PWM_CALIB_STATUS1);
404 if ((val_n > 3) && (val_p > 3)) {
405 delay_val = TRGM_PWM_DELAY_CFG_DELAY_CHAN_CALIB_N_SW_SET(val_n) |
406 TRGM_PWM_DELAY_CFG_DELAY_CHAN_CALIB_P_SW_SET(val_p);
407 trgm->PWM_DELAY_CFG = delay_val;
408 *status = trgm_pwmv2_calibration_mode_end;
409 } else {
410 *status = trgm_pwmv2_calibration_mode_begin;
411 }
412#endif
413#if defined(HPM_IP_FEATURE_TRGM_HRPWM_CALIBRATION_1)
414 delay_val = TRGM_PWM_CALIB_STATUS0_CALIB_RESULT_GET(trgm->PWM_CALIB_STATUS0);
415 if (last_val == 0U) {
416 if (delay_val > 10) {
417 last_val = delay_val;
418 trgm->PWM_DELAY_CFG = delay_val;
419 *status = trgm_pwmv2_calibration_mode_end;
420 } else {
421 *status = trgm_pwmv2_calibration_mode_begin;
422 }
423 } else {
424 diff_val = ((delay_val - last_val) > 0) ? (delay_val - last_val) : (last_val - delay_val);
425
426 if (((float)diff_val / last_val) > 0.25f) {
427 trgm->PWM_DELAY_CFG = delay_val;
428 last_val = delay_val;
429 *status = trgm_pwmv2_calibration_mode_end;
430 } else {
431 *status = trgm_pwmv2_calibration_mode_begin;
432 }
433 }
434#endif
435 }
436 break;
437 case trgm_pwmv2_calibration_mode_end:
438 default:
439 break;
440 }
441}
442#endif
