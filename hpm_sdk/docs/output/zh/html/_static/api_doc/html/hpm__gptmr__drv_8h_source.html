249static inline uint32_t gptmr_channel_get_counter(GPTMR_Type *ptr,
250 uint8_t ch_index,
251 gptmr_counter_type_t capture)
252{
253 uint32_t value;
254 switch (capture) {
255 case gptmr_counter_type_rising_edge:
256 value = (ptr->CHANNEL[ch_index].CAPPOS & GPTMR_CHANNEL_CAPPOS_CAPPOS_MASK) >> GPTMR_CHANNEL_CAPPOS_CAPPOS_SHIFT;
257 break;
258 case gptmr_counter_type_falling_edge:
259 value = (ptr->CHANNEL[ch_index].CAPNEG & GPTMR_CHANNEL_CAPNEG_CAPNEG_MASK) >> GPTMR_CHANNEL_CAPNEG_CAPNEG_SHIFT;
260 break;
261 case gptmr_counter_type_measured_period:
262 value = (ptr->CHANNEL[ch_index].CAPPRD & GPTMR_CHANNEL_CAPPRD_CAPPRD_MASK) >> GPTMR_CHANNEL_CAPPRD_CAPPRD_SHIFT;
263 break;
264 case gptmr_counter_type_measured_duty_cycle:
265 value = (ptr->CHANNEL[ch_index].CAPDTY & GPTMR_CHANNEL_CAPDTY_MEAS_HIGH_MASK) >> GPTMR_CHANNEL_CAPDTY_MEAS_HIGH_SHIFT;
266 break;
267 default:
268 value = (ptr->CHANNEL[ch_index].CNT & GPTMR_CHANNEL_CNT_COUNTER_MASK) >> GPTMR_CHANNEL_CNT_COUNTER_SHIFT;
269 break;
270 }
271 return value;
272}
