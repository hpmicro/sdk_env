230static inline uint32_t gptmr_channel_get_counter(GPTMR_Type *ptr,
231 uint8_t ch_index,
232 gptmr_counter_type_t capture)
233{
234 uint32_t value;
235 switch (capture) {
236 case gptmr_counter_type_rising_edge:
237 value = (ptr->CHANNEL[ch_index].CAPPOS & GPTMR_CHANNEL_CAPPOS_CAPPOS_MASK) >> GPTMR_CHANNEL_CAPPOS_CAPPOS_SHIFT;
238 break;
239 case gptmr_counter_type_falling_edge:
240 value = (ptr->CHANNEL[ch_index].CAPNEG & GPTMR_CHANNEL_CAPNEG_CAPNEG_MASK) >> GPTMR_CHANNEL_CAPNEG_CAPNEG_SHIFT;
241 break;
242 case gptmr_counter_type_measured_period:
243 value = (ptr->CHANNEL[ch_index].CAPPRD & GPTMR_CHANNEL_CAPPRD_CAPPRD_MASK) >> GPTMR_CHANNEL_CAPPRD_CAPPRD_SHIFT;
244 break;
245 case gptmr_counter_type_measured_duty_cycle:
246 value = (ptr->CHANNEL[ch_index].CAPDTY & GPTMR_CHANNEL_CAPDTY_MEAS_HIGH_MASK) >> GPTMR_CHANNEL_CAPDTY_MEAS_HIGH_SHIFT;
247 break;
248 default:
249 value = (ptr->CHANNEL[ch_index].CNT & GPTMR_CHANNEL_CNT_COUNTER_MASK) >> GPTMR_CHANNEL_CNT_COUNTER_SHIFT;
250 break;
251 }
252 return value;
253}
