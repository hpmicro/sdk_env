104static inline void tamp_set_ch_enable(TAMP_Type *ptr, uint8_t ch, bool enable)
105{
106 ch >>= 1u;
107 if (enable) {
108 ptr->TAMP[ch].CONTROL |= TAMP_TAMP_CONTROL_ENABLE_MASK;
