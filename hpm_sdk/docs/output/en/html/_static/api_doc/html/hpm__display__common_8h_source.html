185static inline
186 uint8_t display_get_pixel_size_in_bit(display_pixel_format_t format)
187{
188 switch (format) {
189 case display_pixel_format_argb8888:
190 return 32;
191 case display_pixel_format_rgb565:
192 return 16;
193 case display_pixel_format_yuv422:
194 return 16;
195 case display_pixel_format_ycbcr422:
196 return 16;
197 case display_pixel_format_y8:
198 return 8;
199 case display_pixel_format_raw8:
200 return 8;
201 default:
202 return 0;
203 }
204}
205
213static inline bool display_pixel_format_is_yuv_format(display_pixel_format_t format)
214{
215 switch (format) {
216 case display_pixel_format_yuv422:
217 return true;
218 case display_pixel_format_ycbcr422:
219 return true;
220 default:
221 return false;
222 }
223}
224
232static inline
233 uint8_t display_get_pixel_size_in_byte(display_pixel_format_t format)
234{
235 return display_get_pixel_size_in_bit(format) >> 3;
236}
237
246static inline
247 uint32_t display_get_pitch_length_in_byte(display_pixel_format_t format,
248 uint32_t width_in_pixel)
249{
250 return width_in_pixel * (display_get_pixel_size_in_bit(format) >> 3);
251}
