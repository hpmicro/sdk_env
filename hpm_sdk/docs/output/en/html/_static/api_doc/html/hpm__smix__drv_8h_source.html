237static inline bool smix_dma_check_transfer_complete(SMIX_Type *ptr, uint8_t ch_index)
238{
239 if ((SMIX_DMAC_TC_ST_CH_GET(ptr->DMAC_TC_ST) & (1U << ch_index)) != 0) {
240 ptr->DMAC_TC_ST = (1U << ch_index); /* W1C clear status*/
241 return true;
242 }
243 return false;
244}

256 ptr->DMAC_ABRT_ST = (1U << ch_index); /* W1C clear status*/
257 return true;
258 }
259 return false;
260}
