206static inline void pwm_deinit(PWM_Type *pwm_x)
207{
208 pwm_x->IRQEN = 0;
209 pwm_x->DMAEN = 0;
210 pwm_x->SR |= pwm_x->SR;
211 pwm_x->STA = 0;
212 pwm_x->RLD = PWM_RLD_RLD_MASK;
213 for (uint8_t i = 0; i < PWM_SOC_CMP_MAX_COUNT; i++) {
214 pwm_x->CMP[i] = PWM_CMP_CMP_MASK;
215 pwm_x->CMPCFG[i] = 0;
216 pwm_x->CHCFG[i] = PWM_CHCFG_CMPSELEND_SET(PWM_SOC_CMP_MAX_COUNT - 1) | PWM_CHCFG_CMPSELBEG_SET(PWM_SOC_CMP_MAX_COUNT - 1);
217 }
218 pwm_x->FRCMD = 0;
219 pwm_x->GCR = 0;
220 pwm_x->SHCR = 0;
221#if defined(PWM_SOC_HRPWM_SUPPORT) && PWM_SOC_HRPWM_SUPPORT
222 pwm_x->HRPWM_CFG = 0;
223#endif
224 for (uint8_t i = 0; i < PWM_SOC_OUTPUT_TO_PWM_MAX_COUNT; i++) {
225 pwm_x->PWMCFG[i] = 0;
226 }
227}

235{
236 if (pwm_x->SHCR & PWM_SHCR_SHLKEN_MASK) {

645static inline void pwm_config_cmp(PWM_Type *pwm_x, uint8_t index, pwm_cmp_config_t *config)
646{
647 pwm_shadow_register_unlock(pwm_x);
648 if (config->mode == pwm_cmp_mode_output_compare) {
649#if defined(PWM_SOC_HRPWM_SUPPORT) && PWM_SOC_HRPWM_SUPPORT
650 if (config->enable_hrcmp) {
651 pwm_x->CMPCFG[index] = PWM_CMPCFG_CMPSHDWUPT_SET(config->update_trigger);
652 pwm_x->CMP[index] = PWM_CMP_HRPWM_CMP_SET(config->cmp)
653 | PWM_CMP_HRPWM_CMP_HR_SET(config->hrcmp);
654 } else {
655#endif
656 pwm_x->CMPCFG[index] = (config->enable_ex_cmp ? PWM_CMPCFG_XCNTCMPEN_MASK : 0)
657 | PWM_CMPCFG_CMPSHDWUPT_SET(config->update_trigger);
658 pwm_x->CMP[index] = PWM_CMP_CMP_SET(config->cmp)
659 | PWM_CMP_XCMP_SET(config->ex_cmp)
660 | PWM_CMP_CMPHLF_SET(config->half_clock_cmp)
661 | PWM_CMP_CMPJIT_SET(config->jitter_cmp);
662#if defined(PWM_SOC_HRPWM_SUPPORT) && PWM_SOC_HRPWM_SUPPORT
663 }
664#endif
665 } else {
666 pwm_x->CMPCFG[index] |= PWM_CMPCFG_CMPMODE_MASK;
667 }
668}
