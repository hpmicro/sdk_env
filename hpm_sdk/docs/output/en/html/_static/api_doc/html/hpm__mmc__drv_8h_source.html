189static inline void mmc_set_sysclk_freq(MMC_Type *base, uint32_t freq)
190{
191 uint32_t period;
192 base->SYSCLK_FREQ = freq;
193 /* 1/freq *(2^24)*(2^20) */
194 period = (uint32_t)((double)(1 << 20) * (1 << 24) / freq);
195 base->SYSCLK_PERIOD = period;
196}
