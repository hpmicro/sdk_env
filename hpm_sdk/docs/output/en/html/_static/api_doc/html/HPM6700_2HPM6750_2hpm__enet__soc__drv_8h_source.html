16static inline hpm_stat_t enet_intf_selection(ENET_Type *ptr, uint8_t inf_type)
17{
18 hpm_stat_t stat = status_success;
19
20 if (ptr == HPM_ENET0) {
21 HPM_CONCTL->CTRL2 &= ~CONCTL_CTRL2_ENET0_PHY_INTF_SEL_MASK;
22 HPM_CONCTL->CTRL2 |= CONCTL_CTRL2_ENET0_PHY_INTF_SEL_SET(inf_type);
23 } else if (ptr == HPM_ENET1) {
24 HPM_CONCTL->CTRL3 &= ~CONCTL_CTRL3_ENET1_PHY_INTF_SEL_MASK;
25 HPM_CONCTL->CTRL3 |= CONCTL_CTRL3_ENET1_PHY_INTF_SEL_SET(inf_type);

80static inline hpm_stat_t enet_rmii_enable_clock(ENET_Type *ptr, bool internal)
81{
82 hpm_stat_t stat = status_success;
83
84 /* use an internal PLL clock as reference clock for rmii mode */
85 if (ptr == HPM_ENET0) {
86 if (internal == true) {
87 /* use a pll clock */
88 HPM_CONCTL->CTRL2 |= CONCTL_CTRL2_ENET0_REFCLK_OE_MASK | CONCTL_CTRL2_ENET0_RMII_TXCLK_SEL_MASK;
89 } else {
90 /* use an external clock as reference clock for rmii mode */
91 HPM_CONCTL->CTRL2 |= CONCTL_CTRL2_ENET0_RMII_TXCLK_SEL_MASK; /* use an external clock */
92 }
93 } else if (ptr == HPM_ENET1) {
94 if (internal == true) {
95 /* use a pll clock */
96 HPM_CONCTL->CTRL3 |= CONCTL_CTRL3_ENET1_REFCLK_OE_MASK | CONCTL_CTRL3_ENET1_RMII_TXCLK_SEL_MASK;
97 } else {
98 /* use an external clock as reference clock for rmii mode */
99 HPM_CONCTL->CTRL3 |= CONCTL_CTRL3_ENET1_RMII_TXCLK_SEL_MASK; /* use an external clock */
100 }
101 } else {
102 return status_invalid_argument;
103 }
104
105 return stat;
106}
