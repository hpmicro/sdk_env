93static inline void rtc_enable_alarm_interrupt(RTC_Type *base, uint32_t index, bool enable)
94{
95 if (index > 1) {
96 return;
97 }
98
99 uint32_t mask = (index == 0U) ? RTC_ALARM_EN_ENABLE0_MASK : RTC_ALARM_EN_ENABLE1_MASK;

101 if (enable) {
