273static inline uint32_t count_set_bits(uint32_t value)
274{
275 if (value == 0) {

277 }

289static inline uint32_t get_first_set_bit_from_lsb(uint32_t value)
290{
291 uint32_t i = 0;
292 if (!value) {
293 return 0xFFFFFFFFUL;
294 }
295 while (value && !(value & 0x1)) {
296 value >>= 1;
297 i++;
298 }
299 return i;
300}

310static inline uint32_t get_first_set_bit_from_msb(uint32_t value)
311{
312 uint32_t i = 31;
313 if (!value) {
314 return 0xFFFFFFFFUL;
315 }
316 while (value && !(value & 0x80000000)) {
317 value <<= 1;
318 value &= ~1;
319 i--;
320 }
321 return i;
322}
