87/*
88 * DAC_MODE (WO)
89 *
90 * 00: direct mode, DAC output the fixed configured data(from sw_dac_data)
91 * 01: step mode, DAC output from start_point to end point, with configured step, can step up or step down
92 * 10: buffer mode, read data from buffer, then output to analog, internal DMA will load next burst if enough space in local FIFO;
93 * 11: trigger mode, DAC output from external trigger signals
94 * Note:
95 * Trigger mode is not supported in hpm63xx and hpm62xx families.
96 */

98#define DAC_CFG0_DAC_MODE_SHIFT (4U)
