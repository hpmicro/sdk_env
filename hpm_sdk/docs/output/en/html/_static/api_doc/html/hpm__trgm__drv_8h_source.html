103static inline void trgm_input_filter_set_filter_length(TRGM_Type *ptr, uint8_t input, uint32_t length)
104{
105#if defined(TRGM_SOC_HAS_FILTER_SHIFT) && TRGM_SOC_HAS_FILTER_SHIFT
106 uint32_t len = length;
107 uint8_t shift;
108 for (shift = 0; shift <= (TRGM_FILTCFG_FILTLEN_SHIFT_MASK >> TRGM_FILTCFG_FILTLEN_SHIFT_SHIFT); shift++) {
109 if (shift > 0) {
110 len >>= 1u;
111 }
112 if (len <= (TRGM_FILTCFG_FILTLEN_BASE_MASK >> TRGM_FILTCFG_FILTLEN_BASE_SHIFT)) {
113 break;
114 }
115 }
116 if (len > (TRGM_FILTCFG_FILTLEN_BASE_MASK >> TRGM_FILTCFG_FILTLEN_BASE_SHIFT)) {
117 len = (TRGM_FILTCFG_FILTLEN_BASE_MASK >> TRGM_FILTCFG_FILTLEN_BASE_SHIFT);
118 shift = (TRGM_FILTCFG_FILTLEN_SHIFT_MASK >> TRGM_FILTCFG_FILTLEN_SHIFT_SHIFT);
119 }
120 ptr->FILTCFG[input] = (ptr->FILTCFG[input] & ~(TRGM_FILTCFG_FILTLEN_BASE_MASK | TRGM_FILTCFG_FILTLEN_SHIFT_MASK))
121 | TRGM_FILTCFG_FILTLEN_BASE_SET(len) | TRGM_FILTCFG_FILTLEN_SHIFT_SET(shift);
122#else
123 ptr->FILTCFG[input] = (ptr->FILTCFG[input] & ~TRGM_FILTCFG_FILTLEN_MASK)
124 | TRGM_FILTCFG_FILTLEN_SET(length);
125#endif
126}

262static inline void trgm_pwmv2_calibrate_delay_chain(TRGM_Type *trgm, trgm_pwmv2_calibration_mode_t *status)
263{
264#if defined(HPM_IP_FEATURE_TRGM_HRPWM_CALIBRATION_2)
265 uint16_t val_n, val_p;
266#endif
267 uint32_t delay_val;
268#if defined(HPM_IP_FEATURE_TRGM_HRPWM_CALIBRATION_1)
269 static uint16_t last_val = 0;
270 int16_t diff_val;
271#endif
272
273 switch (*status) {
274 case trgm_pwmv2_calibration_mode_begin:
275 trgm->PWM_CALIB_CFG &= ~TRGM_PWM_CALIB_CFG_CALIB_HW_ENABLE_MASK;
276 trgm->PWM_CALIB_CFG &= ~TRGM_PWM_CALIB_CFG_CALIB_SW_START_MASK;
277 trgm->PWM_CALIB_CFG |= TRGM_PWM_CALIB_CFG_CALIB_SW_START_MASK;
278 *status = trgm_pwmv2_calibration_mode_wait;
279 break;
280 case trgm_pwmv2_calibration_mode_wait:
281 if (TRGM_PWM_CALIB_STATUS0_CALIB_ON_GET(trgm->PWM_CALIB_STATUS0) == 0U) {
282#if defined(HPM_IP_FEATURE_TRGM_HRPWM_CALIBRATION_2)
283 val_n = TRGM_PWM_CALIB_STATUS1_CALIB_RESULT_N_GET(trgm->PWM_CALIB_STATUS1);
284 val_p = TRGM_PWM_CALIB_STATUS1_CALIB_RESULT_P_GET(trgm->PWM_CALIB_STATUS1);
285 if ((val_n > 3) && (val_p > 3)) {
286 delay_val = TRGM_PWM_DELAY_CFG_DELAY_CHAN_CALIB_N_SW_SET(val_n) |
287 TRGM_PWM_DELAY_CFG_DELAY_CHAN_CALIB_P_SW_SET(val_p);
288 trgm->PWM_DELAY_CFG = delay_val;
289 *status = trgm_pwmv2_calibration_mode_end;
290 } else {
291 *status = trgm_pwmv2_calibration_mode_begin;
292 }
293#endif
294#if defined(HPM_IP_FEATURE_TRGM_HRPWM_CALIBRATION_1)
295 delay_val = TRGM_PWM_CALIB_STATUS0_CALIB_RESULT_GET(trgm->PWM_CALIB_STATUS0);
296 if (last_val == 0U) {
297 if (delay_val > 10) {
298 last_val = delay_val;
299 trgm->PWM_DELAY_CFG = delay_val;
300 *status = trgm_pwmv2_calibration_mode_end;
301 } else {
302 *status = trgm_pwmv2_calibration_mode_begin;
303 }
304 } else {
305 diff_val = ((delay_val - last_val) > 0) ? (delay_val - last_val) : (last_val - delay_val);
306
307 if (((float)diff_val / last_val) > 0.25f) {
308 trgm->PWM_DELAY_CFG = delay_val;
309 last_val = delay_val;
310 *status = trgm_pwmv2_calibration_mode_end;
311 } else {
312 *status = trgm_pwmv2_calibration_mode_begin;
313 }
314 }
315#endif
316 }
317 break;
318 case trgm_pwmv2_calibration_mode_end:
319 default:
320 break;
321 }
322}
323#endif
