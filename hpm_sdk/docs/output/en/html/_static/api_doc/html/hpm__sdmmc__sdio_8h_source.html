499static inline uint32_t sdio_cis_csr_addr(const uint8_t *reg_ptr)
500{
501 uint32_t reg_addr = 0;
502 for (uint32_t i = 0; i < 3; i++) {
503 reg_addr |= ((uint32_t) reg_ptr[i]) << (i * 8);
504 }
505 return reg_addr;
506}
