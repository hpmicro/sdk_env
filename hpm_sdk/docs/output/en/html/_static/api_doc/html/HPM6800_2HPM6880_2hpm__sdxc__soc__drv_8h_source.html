39static inline void sdxc_set_clock_divider(SDXC_Type *base, uint32_t div)
40{
41 base->MISC_CTRL0 =
42 (base->MISC_CTRL0 & ~SDXC_MISC_CTRL0_FREQ_SEL_SW_MASK) | SDXC_MISC_CTRL0_FREQ_SEL_SW_SET(div - 1U) |
43 SDXC_MISC_CTRL0_FREQ_SEL_SW_EN_MASK;

59static inline void sdxc_wait_card_active(SDXC_Type *base)
60{
61 base->SYS_CTRL |= SDXC_SYS_CTRL_SD_CLK_EN_MASK;
62
63 base->MISC_CTRL1 |= SDXC_MISC_CTRL1_CARD_ACTIVE_MASK;
64
65 while (!IS_HPM_BITMASK_SET(base->MISC_CTRL1, SDXC_MISC_CTRL1_CARD_ACTIVE_MASK)) {

130static inline uint32_t sdxc_get_default_strobe_delay(SDXC_Type *base)
131{
132 (void) base;
133 uint32_t num_delaycells = 0;
134 if (IOC_PAD_FUNC_CTL_ALT_SELECT_GET(HPM_IOC->PAD[IOC_PAD_PC00].FUNC_CTL) == IOC_PC00_FUNC_CTL_SDC0_DS) {
135 num_delaycells = 2;
136 }
137 return num_delaycells;
138}

140static inline uint32_t sdxc_get_default_cardclk_delay_chain(SDXC_Type *base, uint32_t clock_freq)
141{
142 (void) base;
143 uint32_t num_delaycells = 3;
144 if (clock_freq <= 52000000) {
145 num_delaycells = 26;
